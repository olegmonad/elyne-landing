public function generateWishScene(
    string $resolvedPrompt,
    string $originalScene,
    string $userWish,
): ?string {
    $contextBlock = <<<TXT
    ============== ТЕКУЩАЯ СЦЕНА (FAL-промпт) ==============
    {$originalScene}

    ============== ПОЖЕЛАНИЕ ПОЛЬЗОВАТЕЛЯ ==============
    {$userWish}

    ============== ИНСТРУКЦИЯ ==============
    Встрой пожелание в сцену. Не цитируй его буквально.
    Описывай локацию, свет, одежду, время суток ЯВНО.
    Верни самодостаточный FAL-промпт.
    ============== МЕТАПРОМПТ ==============


    TXT;

    $result = $this->callJsonStrict(
        userPrompt: $contextBlock . $resolvedPrompt,
        maxTokens: 2000,
        maxRetries: 2,
        validator: fn ($r) => is_array($r)
            && isset($r['detailed_description'])
            && is_string($r['detailed_description'])
            && trim($r['detailed_description']) !== '',
    );

    return $result['detailed_description'] ?? null;
}

5. Новый файл app/Jobs/InterpretWishAndGenerateJob.php

<?php

namespace App\Jobs;

use App\Contracts\TextAiServiceInterface;
use App\Enums\GenerationPromptType;
use App\Enums\GenerationStatus;
use App\Models\GenerationPrompt;
use App\Models\GenerationRequest;
use App\Models\GenerationResult;
use App\Models\Sphere;
use App\Models\UserRelative;
use App\Services\PromptGenerationService;
use App\Services\PromptPlaceholderService;
use Illuminate\Foundation\Queue\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Support\Facades\Log;

/**
 * Async wish interpretation for sphere regen with user comment.
 *
 * Replaces the direct buildCommentOnlyPrompt() → FAL path (Branch A) in
 * DispatchSpherePhotoAction. Routes wish through OpenRouter first so the
 * "Я иду у Кайласа" style inputs produce a proper cinematic scene instead
 * of a raw literal prompt.
 *
 * Flow: wish + iter=1 scene + profile → OpenRouter → enriched scene → FAL
 */
class InterpretWishAndGenerateJob implements ShouldQueue
{
    use Queueable;

    public int $tries = 2;
    public array $backoff = [10, 30];

    public function __construct(
        public readonly int $generationRequestId,
        public readonly int $generationResultId,
    ) {}

    public function handle(
        TextAiServiceInterface $ai,
        PromptPlaceholderService $placeholderService,
    ): void {
        $request = GenerationRequest::with(['userScene.card.user', 'userScene.scene'])
            ->find($this->generationRequestId);

        if (! $request) {
            Log::warning('InterpretWishAndGenerateJob: request not found', ['id' => $this->generationRequestId]);
            return;
        }

        // Idempotency guard
        if (filled($request->final_prompt_text)) {
            return;
        }

        $userScene = $request->userScene;
        $card = $userScene->card;
        $sphere = Sphere::where('field_number', $userScene->field_number)->first();

        // Fetch the wish interpretation prompt from admin
        $wishPrompt = GenerationPrompt::query()
            ->where('type', GenerationPromptType::WishScene->value)
            ->whereNull('field_number')
            ->first();

        if (! $wishPrompt) {
            Log::error('InterpretWishAndGenerateJob: wish_scene prompt not configured in admin');
            $this->markFailed();
            return;
        }

        // Get iter=1 scene as narrative anchor
        $iter1 = GenerationRequest::where('user_scene_id', $userScene->id)
            ->where('iteration', 1)
            ->first();
        $originalScene = $iter1?->final_prompt_text
            ?? $iter1?->scene_prompt_text
            ?? $userScene->scene?->detailed_description
            ?? '';

        // Resolve placeholders in the wish prompt
        $resolvedPrompt = $sphere
            ? $placeholderService->resolveForSphere($wishPrompt->meta_prompt, $card, $sphere)
            : $placeholderService->resolve($wishPrompt->meta_prompt, $card);

        // Call OpenRouter
        $enrichedScene = $ai->generateWishScene(
            resolvedPrompt: $resolvedPrompt,
            originalScene: $originalScene,
            userWish: $request->user_comment ?? '',
        );

        if ($enrichedScene === null) {
            Log::error('InterpretWishAndGenerateJob: AI returned null', [
                'request_id' => $this->generationRequestId,
            ]);
            $this->markFailed();
            return;
        }

        // Build final FAL prompt
        $finalPrompt = $enrichedScene;

        // Append image instructions + clothing block (same as DispatchSpherePhotoAction branch C)
        $finalPrompt .= "\n\n" . PromptGenerationService::getImageInstructions();

        if ($sphere) {
            $clothingBlock = $placeholderService->formatChosenClothingStyleBlockForImage($card, $sphere);
            if ($clothingBlock !== '') {
                $finalPrompt .= $clothingBlock;
            }
        }

        // Persist
        $request->update([
            'scene_prompt_text' => $enrichedScene,
            'meta_prompt_text'  => $resolvedPrompt,
            'final_prompt_text' => $finalPrompt,
        ]);

        // Dispatch image generation
        GenerateImageJob::dispatch(
            generationRequestId: $request->id,
            generationResultId: $this->generationResultId,
            promptText: $finalPrompt,
            userPhotoUrl: $card->photo_url,
        )->onQueue('images');
    }

    public function failed(?\Throwable $exception): void
    {
        Log::error('InterpretWishAndGenerateJob: job failed', [
            'request_id' => $this->generationRequestId,
            'error' => $exception?->getMessage(),
        ]);
        $this->markFailed();
    }

    private function markFailed(): void
    {
        GenerationResult::where('id', $this->generationResultId)
            ->update(['status' => GenerationStatus::Failed->value]);
    }
}
